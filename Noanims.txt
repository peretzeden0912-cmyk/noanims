local Players = game:GetService("Players")
local lp = Players.LocalPlayer

local function disableAnim(char)
	task.spawn(function()
		while char.Parent do
			local anim = char:FindFirstChild("Animate")
			if anim and not anim.Disabled then
				pcall(function() anim.Disabled = true end)
			end
			task.wait(1) -- keep checking every second
		end
	end)
end

lp.CharacterAdded:Connect(disableAnim)
if lp.Character then
	disableAnim(lp.Character)
end
